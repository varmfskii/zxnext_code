Layer 2 , Write only overlaid on ROM 

p_layer2: defl $123b
r_layer2: defb $70
start:
  nextreg r_layer2,$10 ; 320x256x256
  ld bc,p_layer2
  ld a,$03       ; enable, wo, 1st 16k
  out (c),a
  ld a,$10       ; bank offset +0
  out (c),a
  call wrtpage
  ld bc,p_layer2
  ld a,$11       ; bank offset +1
  out (c),a
  call wrtpage
  ld bc,p_layer2
  ld a,$12       ; bank offset +2
  out (c),a
  call wrtpage
  ld bc,p_layer2
  ld a,$12       ; bank offset +3
  out (c),a
  call wrtpage
  ld bc,p_layer2
  ld a,$12       ; bank offset +4
  out (c),a
  call wrtpage
  ret
wrtpage:  
  ld hl,$0000
  ld bc,$0040    ; 40*256 writes
loop:
  ld (hl),b
  inc hl
  djnz loop
  dec c
  jr nz,loop
